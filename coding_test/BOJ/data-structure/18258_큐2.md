class Node {
  constructor(value) {
    this.value = value;
    this.prev = null;
    this.next = null;
  }
}

class LinkedList {
  constructor() {
    this.head = null;
    this.tail = null;
    this.length = 0;
  }

  getLength() {
    return this.length;
  }

  empty() {
    return this.length === 0 ? 1 : 0;
  }

  front() {
    const value = this.head.value;

    if(value) return value;
    return -1;
  }

  back() {
    const value = this.tail.value;

    if(value) return value;
    return -1;
  }  

  push(value) {
    const newNode = new Node(value);

    if(this.getLength() === 0) {
      this.head = newNode;
      this.tail = newNode;
    }

    else {
      this.tail.next = newNode;
      newNode.prev = this.tail
      this.tail = newNode;
    }
    this.length++;
  }

  pop() {
    if(this.getLength() === 0) {
      return -1;
    }
    else {
      const result = this.head.value;
      this.head = this.head.next;      

      this.length--;
      return result;
    }
  }  
}

const solution = (N, rest) => {
  const linkedList = new LinkedList();
  const answer = [];

  rest.forEach(el => {
    switch(el) {
      case 'pop' : {
        const value = linkedList.pop();
        answer.push(value);
        break;
      }

      case 'size' :
        const value = linkedList.getLength();
        answer.push(value);
        break;

      case 'empty' : {
        const value = linkedList.empty();
        answer.push(value);
        break;
      }

      case 'front' : {
        const value = linkedList.front();
        answer.push(value);
        break;
      }

      case 'back' : {
        const value = linkedList.back();
        answer.push(value);
        break;
      }   

      default : {
        const value = el.split(' ')[1];
        linkedList.push(value);
        break;
      }
    }
  })

  return answer.join('\n');
}

const [ N, ...rest ] = require('fs').readFileSync('/dev/stdin').toString().trim().split('\n');

console.log(solution(Number(N), rest));
